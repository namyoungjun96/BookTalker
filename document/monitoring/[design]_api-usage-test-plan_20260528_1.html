// 스프링 컨텍스트 불필요 — 순수 단위 테스트
class ApiUsageCounterTest {

    private ApiUsageCounter counter;

    @BeforeEach
    void setUp() { counter = new ApiUsageCounter(); }

    @Test
    void getTopN_내림차순_정렬() {
        counter.increment("GET /review"); counter.increment("GET /review"); counter.increment("GET /review");
        counter.increment("GET /rank");   counter.increment("GET /rank");
        counter.increment("GET /book");

        List<Map.Entry<String, Long>> top = counter.getTopN(3);

        assertThat(top.get(0).getKey()).isEqualTo("GET /review");
        assertThat(top.get(0).getValue()).isEqualTo(3L);
        assertThat(top.get(1).getKey()).isEqualTo("GET /rank");
    }
}
단위 테스트 ApiUsageInterceptorTest.java 테스트 대상: ApiUsageInterceptor.java
테스트 메서드givenexpected
preHandle_HandlerMethod일때_카운터_증가 handler = HandlerMethod, 요청 속성에 pattern = "/rank" counter.increment("GET /rank") 1회 호출
preHandle_HandlerMethod_아닐때_스킵 handler = String 객체 (정적 리소스 등) counter.increment() 미호출, true 반환
preHandle_패턴_없을때_URI_폴백 BEST_MATCHING_PATTERN_ATTRIBUTE = null, URI = "/rank" counter.increment("GET /rank") 호출 
@ExtendWith(MockitoExtension.class)
class ApiUsageInterceptorTest {

    @Mock ApiUsageCounter counter;
    @Mock HttpServletRequest request;
    @Mock HttpServletResponse response;
    @Mock HandlerMethod handlerMethod;
    @InjectMocks ApiUsageInterceptor interceptor;

    @Test
    void preHandle_HandlerMethod일때_카운터_증가() {
        given(request.getMethod()).willReturn("GET");
        given(request.getAttribute(HandlerMapping.BEST_MATCHING_PATTERN_ATTRIBUTE))
            .willReturn("/rank");

        boolean result = interceptor.preHandle(request, response, handlerMethod);

        assertThat(result).isTrue();
        then(counter).should().increment("GET /rank");
    }

    @Test
    void preHandle_HandlerMethod_아닐때_스킵() {
        boolean result = interceptor.preHandle(request, response, "not-a-handler");

        assertThat(result).isTrue();
        then(counter).shouldHaveNoInteractions();
    }
}

buildReportMessage_상위3개_포함 counter에 3개 키 주입 (각 호출 수 다름) 메시지에 "1위", "2위", "3위" 포함, 호출 수 내림차순

sendDailyReport_실행후_카운터_초기화 counter에 데이터 있음, webhook URL 미설정 실행 후 counter.getTopN(1) → 빈 리스트

⚠ Discord 실제 전송 방지: 테스트 프로퍼티에서 app.discord.webhook-url을 빈 값으로 설정.
webhook URL이 없으면 sendToDiscord()가 경고 로그만 남기고 반환하므로 외부 호출 없음. 

@SpringBootTest
@ActiveProfiles("local")
@TestPropertySource(properties = "app.discord.webhook-url=")  // Discord 전송 차단
class ApiUsageReportSchedulerTest {

    @Autowired ApiUsageReportScheduler scheduler;
    @Autowired ApiUsageCounter counter;

    @BeforeEach
    void setUp() { counter.clear(); }

    @Test
    void buildReportMessage_상위3개_포함() {
        counter.increment("GET /rank"); counter.increment("GET /rank");
        counter.increment("GET /book/search");
        counter.increment("GET /review/list"); counter.increment("GET /review/list"); counter.increment("GET /review/list");

        String msg = scheduler.buildReportMessage();  // package-private으로 변경 후 접근 가능

        assertThat(msg).contains("1위", "GET /review/list");
        assertThat(msg).contains("3회");
    }

    @Test
    void sendDailyReport_실행후_카운터_초기화() {
        counter.increment("GET /rank");

        scheduler.sendDailyReport();

        assertThat(counter.getTopN(1)).isEmpty();
    }
}
